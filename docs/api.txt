/account: [GET] return current user's personal information

/account/login: [POST]
    param: [email, password]
    return: "ACCOUNT_LOGIN_SUCCESS" / "ACCOUNT_LOGIN_INVALID"

/account/register: [POST]
    param: [user_id, email, password, name, address, phone, is_expert]
    return: "ACCOUNT_REGISTER_SUCCESS" / "ACCOUNT_REGISTER_EXIST"

/account/set: [POST] set current user's personal informations
    param: [(email), (password), (name), (address), (phone),
           (is_expert), (is_expert_verified), (expert_description), (expert_experience), (expert_location)]
    return: "ACCOUNT_SET_SUCCESS"

/account/logout: [GET]
    return: "ACCOUNT_LOGOUT_SUCCESS"

--------------------------------

/appointment: [GET] return current user's appointments

/appointment/add: [POST] add an appointment
    param: [expert_id, timestamp, location, issue_description, status]
    return: "APPOINTMENT_ADD_SUCCESS"

/appointment/set: [POST] set appointment status
    param: [patient_id/expert_id, status]
    return: "APPOINTMENT_SET_SUCCESS"

--------------------------------

/chat: [GET] return current user's chats with sender_id
    param: [sender_id]

/chat/send: [POST] send chat between sender and receiver
    param: [receiver_id, message]
    return: "CHAT_SEND_SUCCESS"

--------------------------------

+/detection: [POST] return current user's depression detection

--------------------------------

/monitoring: [GET] return current user's monitoring

--------------------------------

/socmed: [GET] return current user's social media usernames

/socmed/add: [POST] add current user's social media usernames
    param: [(twitter), (facebook), (instagram)]
    return: "SOCMED_ADD_SUCCESS"

/socmed/set: [POST] set current user's social media usernames
    param: [(twitter), (facebook), (instagram)]
    return: "SOCMED_SET_SUCCESS"
